\begin{aligned}
d = \norm{\vec{P} - \vec{Q}}\\
       = \norm{\begin{pmatrix}
      1\\ 
      -3\\
      4\\
  
    \end{pmatrix} - \begin{pmatrix}
      -4\\ 
      1\\
      2\\
  
    \end{pmatrix}} = \norm{\begin{pmatrix}
      5\\ 
      -4\\
      2\\
  
    \end{pmatrix}}
\\
\implies d = \sqrt{5^2 + (-4)^2 + 2^2}\\
 = 3\sqrt{5}
\end{aligned}
\end{equation}
