\begin{problem}
\label{ch4_sim}
Plot 
\begin{equation}
P_e = \pr{\hat{X} = -1|X=1}
\end{equation}
%
for 
\begin{equation}
Y = AX+N,
\end{equation}
where $A$ is Raleigh with $E\sbrak{A^2} = \gamma, N \sim \gauss{0}{1}, X \in \brak{-1,1}$ for $0 \le \gamma \le 10$ dB.
\end{problem}
%
\begin{problem}
Assuming that $N$ is a constant, find an expression for $P_e$.  Call this $P_e(N)$
\end{problem}

\end{problem}
